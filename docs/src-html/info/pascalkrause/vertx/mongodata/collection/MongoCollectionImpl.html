001package info.pascalkrause.vertx.mongodata.collection;
002
003import java.util.Collection;
004import java.util.List;
005import java.util.function.Function;
006import java.util.stream.Collectors;
007
008import info.pascalkrause.vertx.mongodata.SimpleAsyncResult;
009import info.pascalkrause.vertx.mongodata.datasource.MongoDataSource;
010import io.vertx.core.AsyncResult;
011import io.vertx.core.Handler;
012import io.vertx.core.json.JsonObject;
013
014public class MongoCollectionImpl<T> implements MongoCollection<T> {
015
016    private final MongoDataSource mds;
017    private final Function<T, JsonObject> encode;
018    private final Function<JsonObject, T> decode;
019    private final String collectionName;
020
021    /**
022     * <b>Important:</b> When implementing the transformations, check that you use the key "_id" for the id. For binary
023     * types you should use a new JsonObject which contains a key with the name "$binary".
024     * <p>
025     * Example:<br>
026     * 
027     * <pre>
028     * {
029     *     JsonObject encoded = new JsonObject();
030     *     encoded.put("_id", resource.getId());
031     *     encoded.put("image", new JsonObject().put("$binary", resource.getImageBytes()));
032     * }
033     * </pre>
034     * 
035     * @param collectionName The name of the collection in the database.
036     * @param encode A method to transform the resource object into a JsonObject which can be handled by the database.
037     * @param decode A method to transform the JsonObject return from the database into a resource object.
038     * @param client An instance of MongoClient which is used to talk with the Mongo database.
039     */
040    public MongoCollectionImpl(String collectionName, Function<T, JsonObject> encode, Function<JsonObject, T> decode,
041            MongoDataSource mds) {
042        this.collectionName = collectionName;
043        this.encode = encode;
044        this.decode = decode;
045        this.mds = mds;
046    }
047
048    @Override
049    public String getCollectionName() {
050        return collectionName;
051    }
052
053    @Override
054    public MongoCollectionImpl<T> countAll(Handler<AsyncResult<Long>> resultHandler) {
055        return count(new JsonObject(), resultHandler);
056    }
057
058    @Override
059    public MongoCollectionImpl<T> count(JsonObject query, Handler<AsyncResult<Long>> resultHandler) {
060        mds.count(collectionName, query, resultHandler);
061        return this;
062    }
063
064    @Override
065    public MongoCollectionImpl<T> bulkInsert(Collection<T> resources, Handler<AsyncResult<Long>> resultHandler) {
066        List<JsonObject> documents = resources.parallelStream().map(resource -> encode.apply(resource))
067                .collect(Collectors.toList());
068        mds.bulkInsert(collectionName, documents, resultHandler);
069        return this;
070    }
071
072    @Override
073    public MongoCollectionImpl<T> upsert(T resource, Handler<AsyncResult<String>> resultHandler) {
074        mds.upsert(collectionName, encode.apply(resource), resultHandler);
075        return this;
076    }
077
078    @Override
079    public MongoCollectionImpl<T> findAll(Handler<AsyncResult<List<T>>> resultHandler) {
080        find(new JsonObject(), resultHandler);
081        return this;
082    }
083
084    @Override
085    public MongoCollectionImpl<T> find(JsonObject query, Handler<AsyncResult<List<T>>> resultHandler) {
086        mds.find(collectionName, query, dbResponse -> {
087            if (dbResponse.failed()) {
088                resultHandler.handle(new SimpleAsyncResult<List<T>>(dbResponse.cause()));
089                return;
090            }
091            List<T> results = dbResponse.result().parallelStream().map(json -> decode.apply(json))
092                    .collect(Collectors.toList());
093            resultHandler.handle(new SimpleAsyncResult<List<T>>(results));
094        });
095        return this;
096    }
097
098    @Override
099    public MongoCollectionImpl<T> remove(JsonObject query, Handler<AsyncResult<Long>> resultHandler) {
100        mds.removeDocuments(collectionName, query, resultHandler);
101        return this;
102    }
103
104    @Override
105    public MongoCollectionImpl<T> remove(String id, Handler<AsyncResult<Long>> resultHandler) {
106        JsonObject removeQuery = new JsonObject();
107        removeQuery.put("_id", id);
108        remove(removeQuery, resultHandler);
109        return this;
110    }
111
112    @Override
113    public MongoCollectionImpl<T> drop(Handler<AsyncResult<Void>> resultHandler) {
114        mds.dropCollection(collectionName, resultHandler);
115        return this;
116    }
117}
