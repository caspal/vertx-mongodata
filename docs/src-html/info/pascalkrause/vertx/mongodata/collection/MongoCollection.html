001package info.pascalkrause.vertx.mongodata.collection;
002
003import java.util.Collection;
004import java.util.List;
005
006import io.vertx.core.AsyncResult;
007import io.vertx.core.Handler;
008import io.vertx.core.json.JsonObject;
009
010public interface MongoCollection<T> {
011
012    /**
013     *
014     * @return The name of the collection.
015     */
016    public String getCollectionName();
017
018    /**
019     * Count matching entries in a collection.
020     * 
021     * @param query A filter for the count request. For counting all entries in a collection use
022     * {@link #countAll(Handler)}.
023     * @param resultHandler A handler to handle the AsnycResult which contains the number of matching entries.
024     * @return The actual MongoCollection.
025     */
026    public MongoCollection<T> count(JsonObject query, Handler<AsyncResult<Long>> resultHandler);
027
028    /**
029     * Count all entries in a collection.
030     * 
031     * @param resultHandler A handler to handle the AsnycResult which contains the number of all entries.
032     * @return The actual MongoCollection.
033     */
034    public MongoCollection<T> countAll(Handler<AsyncResult<Long>> resultHandler);
035
036    /**
037     * Finds matching entries in a collection.
038     * 
039     * @param query A filter for the find request. For finding all entries in a collection use
040     * {@link #findAll(Handler)}.
041     * @param resultHandler A handler to handle the AsnycResult which contains a List with all matching entries.
042     * @return The actual MongoCollection.
043     */
044    public MongoCollection<T> find(JsonObject query, Handler<AsyncResult<List<T>>> resultHandler);
045
046    /**
047     * Finds all entries in a collection.
048     * 
049     * @param resultHandler A handler to handle the AsnycResult which contains a List with all entries.
050     * @return The actual MongoCollection.
051     */
052    public MongoCollection<T> findAll(Handler<AsyncResult<List<T>>> resultHandler);
053
054    /**
055     * Insert or update a resource into the collection.
056     * 
057     * @param resource The resource to insert or update.
058     * @param resultHandler A handler to handle the AsnycResult which contains a String with the new id (insert), or
059     * null (update).
060     * @return The actual MongoCollection.
061     */
062    public MongoCollection<T> upsert(T resource, Handler<AsyncResult<String>> resultHandler);
063
064    /**
065     * Inserts many resources into the collection.
066     * 
067     * @param resources A collection of resources to insert into the collection.
068     * @param resultHandler A handler to handle the AsnycResult which contains the number of inserted resources.
069     * @return The actual MongoCollection.
070     */
071    public MongoCollection<T> bulkInsert(Collection<T> resources, Handler<AsyncResult<Long>> resultHandler);
072
073    /**
074     * Removes the resource with the passed ID from the collection.
075     * 
076     * @param id The ID of the resource to delete.
077     * @param resultHandler A handler to handle the AsnycResult which contains the number of deleted entries.
078     * @return The actual MongoCollection.
079     */
080    public MongoCollection<T> remove(String id, Handler<AsyncResult<Long>> resultHandler);
081
082    /**
083     * Remove matching entries in a collection.
084     * 
085     * @param query A filter for the remove request.
086     * @param resultHandler A handler to handle the AsnycResult which contains the number of deleted entries.
087     * @return The actual MongoCollection.
088     */
089    public MongoCollection<T> remove(JsonObject query, Handler<AsyncResult<Long>> resultHandler);
090
091    /**
092     * Drop the collection.
093     * 
094     * @param resultHandler A handler which will be called when complete.
095     * @return The actual MongoCollection.
096     */
097    public MongoCollection<T> drop(Handler<AsyncResult<Void>> resultHandler);
098}
