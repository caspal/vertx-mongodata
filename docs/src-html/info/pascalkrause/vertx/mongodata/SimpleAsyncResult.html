001package info.pascalkrause.vertx.mongodata;
002
003import java.util.Objects;
004
005import io.vertx.core.AsyncResult;
006
007public class SimpleAsyncResult<E> implements AsyncResult<E> {
008
009    private final Throwable cause;
010    private final E result;
011
012    public SimpleAsyncResult(Throwable cause) {
013        this.cause = cause;
014        this.result = null;
015    }
016    
017    public SimpleAsyncResult(E result) {
018        this.cause = null;
019        this.result = result;
020    }
021
022    @Override
023    public E result() {
024        return result;
025    }
026
027    @Override
028    public Throwable cause() {
029        return cause;
030    }
031
032    @Override
033    public boolean succeeded() {
034        return !failed();
035    }
036
037    @Override
038    public boolean failed() {
039        return Objects.nonNull(cause);
040    }
041}
