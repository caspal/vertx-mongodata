001package info.pascalkrause.vertx.mongodata.service;
002
003import java.util.stream.Collectors;
004import java.util.stream.Stream;
005
006import com.google.common.base.Splitter;
007
008public enum MongoServiceErrorCodes {
009    ACTION_NOT_FOUND(404, "Action Not Found"),
010    UNEXPECTED_ERROR(500, "Unexpected Error");
011
012    private final int statuscode;
013    private final String id;
014    private final String description;
015
016    private MongoServiceErrorCodes(int statuscode, String description) {
017        this.statuscode = statuscode;
018        this.description = description;
019
020        Stream<String> descParts = Splitter.on(" ").splitToList(description).stream();
021        // Transform "Resource Already Exist" to "REALEX"
022        String desc = descParts.map(s -> s.substring(0, 2).toUpperCase()).collect(Collectors.joining(""));
023        id = new StringBuilder("[").append(statuscode).append("-").append(desc).append("]").toString();
024    }
025
026    public int getStatusCode() {
027        return statuscode;
028    }
029
030    public String getId() {
031        return id;
032    }
033
034    public String getDescription() {
035        return description;
036    }
037}
