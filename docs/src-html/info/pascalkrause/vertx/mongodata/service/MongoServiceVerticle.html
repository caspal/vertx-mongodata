001package info.pascalkrause.vertx.mongodata.service;
002
003import static info.pascalkrause.vertx.mongodata.service.MongoService.ACTION;
004import static info.pascalkrause.vertx.mongodata.service.MongoService.BULK_INSERT;
005import static info.pascalkrause.vertx.mongodata.service.MongoService.CHECK_AVAILABILITY;
006import static info.pascalkrause.vertx.mongodata.service.MongoService.COLLECTION;
007import static info.pascalkrause.vertx.mongodata.service.MongoService.COUNT;
008import static info.pascalkrause.vertx.mongodata.service.MongoService.DROP;
009import static info.pascalkrause.vertx.mongodata.service.MongoService.FIND;
010import static info.pascalkrause.vertx.mongodata.service.MongoService.REMOVE;
011import static info.pascalkrause.vertx.mongodata.service.MongoService.UPSERT;
012import static info.pascalkrause.vertx.mongodata.service.MongoServiceErrorCodes.ACTION_NOT_FOUND;
013import static info.pascalkrause.vertx.mongodata.service.MongoServiceErrorCodes.UNEXPECTED_ERROR;
014
015import java.util.List;
016import java.util.stream.Collectors;
017
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;
020
021import io.vertx.core.AbstractVerticle;
022import io.vertx.core.Context;
023import io.vertx.core.Future;
024import io.vertx.core.Vertx;
025import io.vertx.core.eventbus.EventBus;
026import io.vertx.core.eventbus.MessageConsumer;
027import io.vertx.core.json.JsonArray;
028import io.vertx.core.json.JsonObject;
029import io.vertx.ext.mongo.BulkOperation;
030import io.vertx.ext.mongo.MongoClient;;
031
032public class MongoServiceVerticle extends AbstractVerticle {
033
034    private final Logger logger = LoggerFactory.getLogger(MongoServiceVerticle.class);
035    private final String serviceAddress;
036
037    private MongoClient mc;
038    private JsonObject mongoConfig;
039    private EventBus eb;
040
041    public MongoServiceVerticle(String serviceAddress, JsonObject mongoConfig) {
042        this.serviceAddress = serviceAddress;
043        this.mongoConfig = mongoConfig;
044    }
045
046    private MessageConsumer<?> initializeEBConsumer() {
047        return eb.consumer(serviceAddress, message -> {
048            String collectionName = message.headers().get(COLLECTION);
049            switch (message.headers().get(ACTION)) {
050            case CHECK_AVAILABILITY:
051                message.reply(null);
052                break;
053            case COUNT:
054                mc.count(collectionName, (JsonObject) message.body(), res -> {
055                    if (res.failed()) {
056                        message.fail(UNEXPECTED_ERROR.getStatusCode(), UNEXPECTED_ERROR.getDescription());
057                        return;
058                    }
059                    message.reply(res.result());
060                });
061                break;
062            case FIND:
063                mc.find(collectionName, (JsonObject) message.body(), res -> {
064                    if (res.failed()) {
065                        message.fail(UNEXPECTED_ERROR.getStatusCode(), UNEXPECTED_ERROR.getDescription());
066                        return;
067                    }
068                    message.reply(new JsonArray(res.result()));
069                });
070                break;
071            case UPSERT:
072                mc.save(collectionName, (JsonObject) message.body(), res -> {
073                    if (res.failed()) {
074                        message.fail(UNEXPECTED_ERROR.getStatusCode(), UNEXPECTED_ERROR.getDescription());
075                        return;
076                    }
077                    message.reply(res.result());
078                });
079                break;
080            case REMOVE:
081                mc.removeDocuments(collectionName, (JsonObject) message.body(), res -> {
082                    if (res.failed()) {
083                        message.fail(UNEXPECTED_ERROR.getStatusCode(), UNEXPECTED_ERROR.getDescription());
084                        return;
085                    }
086                    message.reply(res.result().getRemovedCount());
087                });
088                break;
089            case BULK_INSERT:
090                JsonArray arr = (JsonArray) message.body();
091                List<BulkOperation> operations = arr.stream()
092                        .map(document -> BulkOperation.createInsert(((JsonObject) document).copy()))
093                        .collect(Collectors.toList());
094                mc.bulkWrite(collectionName, operations, res -> {
095                    if (res.failed()) {
096                        message.fail(UNEXPECTED_ERROR.getStatusCode(), UNEXPECTED_ERROR.getDescription());
097                        return;
098                    }
099                    message.reply(res.result().getInsertedCount());
100                });
101                break;
102            case DROP:
103                mc.dropCollection(collectionName, res -> {
104                    if (res.failed()) {
105                        message.fail(UNEXPECTED_ERROR.getStatusCode(), UNEXPECTED_ERROR.getDescription());
106                        return;
107                    }
108                    message.reply(null);
109                });
110                break;
111            default:
112                message.fail(ACTION_NOT_FOUND.getStatusCode(), ACTION_NOT_FOUND.getDescription());
113                break;
114            }
115        });
116    }
117
118    @Override
119    public void init(Vertx vertx, Context context) {
120        super.init(vertx, context);
121        eb = vertx.eventBus();
122        mc = MongoClient.createShared(vertx, mongoConfig);
123        initializeEBConsumer();
124    }
125
126    @Override
127    public void start(Future<Void> startFuture) throws Exception {
128        MessageConsumer<?> consumer = initializeEBConsumer();
129        consumer.completionHandler(res -> {
130            if (res.succeeded()) {
131                String message = "MongoCollectionService is now available under: " + serviceAddress;
132                logger.info(message);
133                startFuture.complete();
134            } else {
135                String message = "Failed to initialize MongoCollectionService";
136                logger.error(message, res.cause());
137                logger.info(message);
138                startFuture.fail(res.cause());
139            }
140        });
141    }
142
143    @Override
144    public void stop() throws Exception {
145        super.stop();
146        mc.close();
147    }
148}
