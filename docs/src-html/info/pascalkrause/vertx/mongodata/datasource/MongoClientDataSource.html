001package info.pascalkrause.vertx.mongodata.datasource;
002
003import java.util.List;
004import java.util.stream.Collectors;
005
006import info.pascalkrause.vertx.mongodata.SimpleAsyncResult;
007import io.vertx.core.AsyncResult;
008import io.vertx.core.Handler;
009import io.vertx.core.json.JsonObject;
010import io.vertx.ext.mongo.BulkOperation;
011import io.vertx.ext.mongo.MongoClient;
012
013public class MongoClientDataSource implements MongoDataSource {
014
015    private final MongoClient mc;
016
017    public MongoClientDataSource(MongoClient client) {
018        mc = client;
019    }
020
021    @Override
022    public MongoDataSource count(String collection, JsonObject query, Handler<AsyncResult<Long>> resultHandler) {
023        mc.count(collection, query, resultHandler);
024        return this;
025    }
026
027    @Override
028    public MongoDataSource find(String collection, JsonObject query,
029            Handler<AsyncResult<List<JsonObject>>> resultHandler) {
030        mc.find(collection, query, resultHandler);
031        return this;
032    }
033
034    @Override
035    public MongoDataSource upsert(String collection, JsonObject document, Handler<AsyncResult<String>> resultHandler) {
036        mc.save(collection, document, resultHandler);
037        return this;
038    }
039
040    @Override
041    public MongoDataSource bulkInsert(String collection, List<JsonObject> documents,
042            Handler<AsyncResult<Long>> resultHandler) {
043        List<BulkOperation> operations = documents.parallelStream()
044                .map(document -> BulkOperation.createInsert(document.copy())).collect(Collectors.toList());
045
046        mc.bulkWrite(collection, operations, dbResponse -> {
047            if (dbResponse.failed()) {
048                resultHandler.handle(new SimpleAsyncResult<Long>(dbResponse.cause()));
049                return;
050            }
051            resultHandler.handle(new SimpleAsyncResult<Long>(dbResponse.result().getInsertedCount()));
052        });
053        return this;
054    }
055
056    @Override
057    public MongoDataSource removeDocuments(String collection, JsonObject query,
058            Handler<AsyncResult<Long>> resultHandler) {
059        mc.removeDocuments(collection, query, deleteResult -> {
060            if (deleteResult.failed()) {
061                resultHandler.handle(new SimpleAsyncResult<Long>(deleteResult.cause()));
062                return;
063            }
064            resultHandler.handle(new SimpleAsyncResult<Long>(deleteResult.result().getRemovedCount()));
065        });
066        return this;
067    }
068
069    @Override
070    public MongoDataSource dropCollection(String collection, Handler<AsyncResult<Void>> resultHandler) {
071        mc.dropCollection(collection, resultHandler);
072        return this;
073    }
074}
